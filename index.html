توثيق الدوال في JavaScript

تحتوي هذه الصفحة على أمثلة تفاعلية لدوال JavaScript مع شرح بسيط لكل دالة. يمكنك تجربة الأمثلة واختبار فهمك بطريقة ممتعة.

دالة الترحيب

هذه الدالة تعرض رسالة ترحيب باستخدام متغير ثابت.

function hello() {
    const name = 'hsoub';
    console.log('مرحباً بالعالم ' + name);
}

اختبار: قم بتعديل الدالة لتعرض اسمك بدلاً من "hsoub".

تصفية المصفوفة

تقوم هذه الدالة بتصفية عناصر المصفوفة حسب نوعها. في هذا المثال، تعرض القيم النصية فقط.

function filterArray() {
    const array = ['ahmed', 1, 'abd', 'ibrahim', 'somya', 123];
    for (let item of array) {
        if (typeof item === 'string') {
            console.log(item);
        }
    }
}

اختبار: أضف عناصر جديدة إلى المصفوفة واختبر الدالة.

دالة الترحيب مع وسيط

تقبل هذه الدالة وسيطاً لإضافة الاسم إلى الرسالة الترحيبية.

function hello(name) {
    console.log('مرحباً بالعالم ' + name);
}

اختبار: قم بتمرير أسماء مختلفة إلى الدالة.

دالة الجمع

تحسب هذه الدالة مجموع رقمين وتعرض النتيجة.

function sum(x, y) {
    console.log(x + y);
}

اختبار: جرب تمرير قيم سالبة أو صفرية.

دالة الجمع مع الإرجاع

تُرجع هذه النسخة من دالة الجمع النتيجة بدلاً من عرضها.

function sum(x, y) {
    return x + y;
}

اختبار: قم بتخزين القيمة المرجعة في متغير جديد واستخدامها.

تصفية المصفوفة مع النوع

تقوم هذه الدالة بتصفية عناصر المصفوفة بناءً على نوع محدد (مثل 'number').

function filterArray(array, type) {
    for (let item of array) {
        if (typeof item === type) {
            console.log(item);
        }
    }
}

اختبار: جرب تمرير أنواع بيانات مختلفة كمعامل.

المعاملات الافتراضية

توضح هذه الدالة كيفية استخدام المعاملات الافتراضية في JavaScript.

function sum(x, y = 1) {
    return x + y;
}

اختبار: جرب استدعاء الدالة بدون تمرير المعامل الثاني.

استخدام الوسائط

تحسب هذه الدالة مجموع جميع الوسائط المُمررة إليها.

function sum() {
    let total = 0;
    for (let num of arguments) {
        total = total + num;
    }
    return total;
}

اختبار: جرب تمرير عدد مختلف من الوسائط.

المتغيرات والفرق بين let و var

يوضح هذا المثال الفرق بين let و var عند تعريف المتغيرات داخل الحلقات.

function runLet() {
    for (let i = 0; i < 5; i++) {
        console.log(i);
    }
    // لا يمكن الوصول إلى i هنا عند استخدام let
}

function runVar() {
    for (var i = 0; i < 5; i++) {
        console.log(i);
    }
    console.log('آخر قيمة لـ i:', i); // يمكن الوصول إلى i هنا عند استخدام var
}

الأعداد الأولية

يتم استخدام دالتين: واحدة للتحقق إذا كان العدد أوليًا، والأخرى لطباعة جميع الأعداد الأولية حتى قيمة معينة.

function isPrime(num) {
    if (num <= 1) return false;
    for (let i = 2; i < num; i++) {
        if (num % i === 0) return false;
    }
    return true;
}

function printPrimes(max) {
    for (let i = 2; i < max; i++) {
        if (isPrime(i)) console.log(i);
    }
}

let maxNum = prompt('أدخل الرقم الأقصى:');
if (maxNum) printPrimes(Number(maxNum));
