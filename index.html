توثيق JavaScript - أمثلة تفاعلية

حذف العناصر من المصفوفة

يمكنك حذف العناصر من المصفوفة باستخدام عدة طرق مثل pop و shift و splice. انظر المثال التالي:

let numb = [1, 2, 3, 4, 5];
const last = numb.pop();
console.log(last);
console.log(numb);

const first = numb.shift();
console.log(first);
console.log(numb);

const middle = numb.splice(1, 2);
console.log(middle);
console.log(numb);

تفريغ المصفوفة

لتفريغ المصفوفة بالكامل، يمكن استخدام عدة طرق مثل التعيين إلى مصفوفة جديدة أو تغيير الطول. المثال التالي يوضح ذلك:

let langs = ['c++', 'php', 'js'];
let clone = langs;
langs = [];

langs.length = 0;
console.log(langs);
console.log(clone);

إضافة عناصر إلى المصفوفة

يمكنك إضافة عناصر إلى المصفوفة باستخدام طرق مثل push و unshift و splice. المثال التالي يوضح كيفية ذلك:

let alphapet = ['c', 'd'];

alphapet[alphapet.length] = 'e';
alphapet.push('f', 'g');
alphapet.unshift('a', 'b');
alphapet.splice(2, 0, 'run', 'stop');
console.log(alphapet);

البحث في المصفوفة

يمكنك استخدام includes و indexOf للبحث عن عناصر في المصفوفة:

let num0 = [120, 3, 5, 55, 43];
console.log(num0.includes(3)); // تحقق إذا كان العنصر 3 موجودًا
console.log(num0.includes(32)); // تحقق إذا كان العنصر 32 موجودًا
console.log(num0.indexOf(5)); // إرجاع فهرس العنصر 5

استخدام التوابع للبحث المتقدم

يمكنك استخدام find و filter و findIndex:

let num0 = [120, 3, 5, 55, 43];
let resu = num0.find(number4 => number4 > 10); // البحث عن أول عنصر أكبر من 10
console.log(resu);

let resuFiltered = num0.filter(number4 => number4 > 10); // إرجاع جميع العناصر الأكبر من 10
console.log(resuFiltered);

let index = num0.findIndex(number4 => number4 > 10); // إرجاع فهرس أول عنصر أكبر من 10
console.log(index);

العبور على عناصر المصفوفة

استخدام حلقة for...of أو forEach للمرور على العناصر:

let langs = ['js', 'c#', 'ruby'];
for (let lang of langs) {
    console.log(lang);
}
langs.forEach((value, index) => console.log(`${index}: ${value}`));

ترتيب المصفوفة

يمكنك ترتيب العناصر باستخدام sort:

let constArray = [12, 1, 4, 2, 53];
constArray.sort((a, b) => a - b); // الترتيب تصاعديًا
console.log(constArray);
