توثيق JavaScript - أمثلة تفاعلية

حذف العناصر من المصفوفة

يمكنك حذف العناصر من المصفوفة باستخدام عدة طرق مثل pop و shift و splice. انظر المثال التالي:

let numb = [1, 2, 3, 4, 5];
const last = numb.pop();
console.log(last);
console.log(numb);

const first = numb.shift();
console.log(first);
console.log(numb);

const middle = numb.splice(1, 2);
console.log(middle);
console.log(numb);

تفريغ المصفوفة

لتفريغ المصفوفة بالكامل، يمكن استخدام عدة طرق مثل التعيين إلى مصفوفة جديدة أو تغيير الطول. المثال التالي يوضح ذلك:

let langs = ['c++', 'php', 'js'];
let clone = langs;
langs = [];

langs.length = 0;
console.log(langs);
console.log(clone);

إضافة عناصر إلى المصفوفة

يمكنك إضافة عناصر إلى المصفوفة باستخدام طرق مثل push و unshift و splice. المثال التالي يوضح كيفية ذلك:

let alphapet = ['c', 'd'];

alphapet[alphapet.length] = 'e';
alphapet.push('f', 'g');
alphapet.unshift('a', 'b');
alphapet.splice(2, 0, 'run', 'stop');
console.log(alphapet);
